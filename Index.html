Google Sheet via Apps Script

Dati locali attivi (439 dipendenti)
Come collegare il Google Sheet (Apps Script)
1. Apri il tuo Google Sheet → Estensioni → Apps Script
2. Incolla questo codice e salva:
function doGet(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  var type = e.parameter.type || "ana";
  var sheetName = (type === "dpi") ? "DPI X MANSIONE" : "Anagrafe aggiornata";
  var s = sheet.getSheetByName(sheetName);
  if (!s) s = sheet.getSheets()[type === "dpi" ? 1 : 0];
  var data = s.getDataRange().getValues();
  var csv = data.map(function(r){ return r.map(function(c){ 
    var v = String(c).replace(/"/g,'""');
    return v.indexOf(",")>=0||v.indexOf('"')>=0 ? '"'+v+'"' : v;
  }).join(","); }).join("\n");
  return ContentService.createTextOutput(csv)
    .setMimeType(ContentService.MimeType.TEXT);
}
3. Clicca Distribuisci → Nuova distribuzione
   Tipo: App web • Accesso: Chiunque
4. Copia l'URL della Web App e incollalo sopra.
